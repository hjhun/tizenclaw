AUX_SOURCE_DIRECTORY(${CMAKE_CURRENT_SOURCE_DIR}
  SRCS)

ADD_EXECUTABLE(${TARGET_TIZEN_VCONF_CLI}
  ${SRCS})

TARGET_INCLUDE_DIRECTORIES(${TARGET_TIZEN_VCONF_CLI} PRIVATE
  "${CMAKE_CURRENT_SOURCE_DIR}/nlohmann"
)

SET_TARGET_PROPERTIES(
  ${TARGET_TIZEN_VCONF_CLI}
  PROPERTIES COMPILE_FLAGS "-fPIE")
SET_TARGET_PROPERTIES(
  ${TARGET_TIZEN_VCONF_CLI}
  PROPERTIES LINK_FLAGS "-pie")

TARGET_INCLUDE_DIRECTORIES(
  ${TARGET_TIZEN_VCONF_CLI} PUBLIC
  ${VCONF_DEPS_INCLUDE_DIRS}
  ${GLIB_DEPS_INCLUDE_DIRS}
  ${DLOG_DEPS_INCLUDE_DIRS}
)

TARGET_LINK_LIBRARIES(
  ${TARGET_TIZEN_VCONF_CLI} PUBLIC
  ${VCONF_DEPS_LIBRARIES}
  -lglib-2.0 -ldlog
)

SET(INSTALL_DIR
  "/opt/usr/share/tizen-tools/cli/${TARGET_TIZEN_VCONF_CLI}")

INSTALL(TARGETS ${TARGET_TIZEN_VCONF_CLI}
  DESTINATION ${INSTALL_DIR})
INSTALL(FILES "${CMAKE_CURRENT_SOURCE_DIR}/tool.md"
  DESTINATION ${INSTALL_DIR})
