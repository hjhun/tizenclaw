AUX_SOURCE_DIRECTORY(${CMAKE_CURRENT_SOURCE_DIR}
  SRCS)

ADD_EXECUTABLE(${TARGET_TIZEN_SENSOR_CLI}
  ${SRCS})

SET_TARGET_PROPERTIES(
  ${TARGET_TIZEN_SENSOR_CLI}
  PROPERTIES COMPILE_FLAGS "-fPIE")
SET_TARGET_PROPERTIES(
  ${TARGET_TIZEN_SENSOR_CLI}
  PROPERTIES LINK_FLAGS "-pie")

TARGET_INCLUDE_DIRECTORIES(
  ${TARGET_TIZEN_SENSOR_CLI} PUBLIC
  ${CAPI_SYSTEM_SENSOR_DEPS_INCLUDE_DIRS}
  ${GLIB_DEPS_INCLUDE_DIRS}
  ${DLOG_DEPS_INCLUDE_DIRS}
)

# Link only the direct libs we need;
# capi-system-sensor pulls in hal-api-sensor
# transitively via pkgconfig, but the lib
# may not exist in emulator build env.
TARGET_LINK_LIBRARIES(
  ${TARGET_TIZEN_SENSOR_CLI} PUBLIC
  -lcapi-system-sensor -lglib-2.0 -ldlog
)

SET(INSTALL_DIR
  "${APP_DATA_DIR}/tools/cli/${TARGET_TIZEN_SENSOR_CLI}")

INSTALL(TARGETS ${TARGET_TIZEN_SENSOR_CLI}
  DESTINATION ${INSTALL_DIR})
INSTALL(FILES "${CMAKE_CURRENT_SOURCE_DIR}/tool.md"
  DESTINATION ${INSTALL_DIR})
