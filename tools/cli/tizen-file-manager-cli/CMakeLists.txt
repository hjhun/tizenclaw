AUX_SOURCE_DIRECTORY(${CMAKE_CURRENT_SOURCE_DIR}
  SRCS)

ADD_EXECUTABLE(${TARGET_TIZEN_FILE_MANAGER_CLI}
  ${SRCS})

SET_TARGET_PROPERTIES(
  ${TARGET_TIZEN_FILE_MANAGER_CLI}
  PROPERTIES COMPILE_FLAGS "-fPIE")
SET_TARGET_PROPERTIES(
  ${TARGET_TIZEN_FILE_MANAGER_CLI}
  PROPERTIES LINK_FLAGS "-pie")

TARGET_INCLUDE_DIRECTORIES(
  ${TARGET_TIZEN_FILE_MANAGER_CLI} PRIVATE
  ${CMAKE_CURRENT_SOURCE_DIR})

SET(INSTALL_DIR
  "/opt/usr/share/tizen-tools/cli/${TARGET_TIZEN_FILE_MANAGER_CLI}")

INSTALL(TARGETS ${TARGET_TIZEN_FILE_MANAGER_CLI}
  DESTINATION ${INSTALL_DIR})
INSTALL(FILES
  "${CMAKE_CURRENT_SOURCE_DIR}/tool.md"
  DESTINATION ${INSTALL_DIR})
