AUX_SOURCE_DIRECTORY(${CMAKE_CURRENT_SOURCE_DIR}
  SRCS)

ADD_EXECUTABLE(${TARGET_TIZEN_MEDIA_CLI}
  ${SRCS})

SET_TARGET_PROPERTIES(
  ${TARGET_TIZEN_MEDIA_CLI}
  PROPERTIES COMPILE_FLAGS "-fPIE")
SET_TARGET_PROPERTIES(
  ${TARGET_TIZEN_MEDIA_CLI}
  PROPERTIES LINK_FLAGS "-pie")

APPLY_PKG_CONFIG(
  ${TARGET_TIZEN_MEDIA_CLI} PUBLIC
  CAPI_CONTENT_MEDIA_CONTENT_DEPS
  CAPI_MEDIA_METADATA_EXTRACTOR_DEPS
  CAPI_CONTENT_MIME_TYPE_DEPS
  GLIB_DEPS
  DLOG_DEPS
)

TARGET_INCLUDE_DIRECTORIES(
  ${TARGET_TIZEN_MEDIA_CLI} PRIVATE
  ${CMAKE_CURRENT_SOURCE_DIR})

SET(INSTALL_DIR
  "/opt/usr/share/tizen-tools/cli/${TARGET_TIZEN_MEDIA_CLI}")

INSTALL(TARGETS ${TARGET_TIZEN_MEDIA_CLI}
  DESTINATION ${INSTALL_DIR})
INSTALL(FILES "${CMAKE_CURRENT_SOURCE_DIR}/tool.md"
  DESTINATION ${INSTALL_DIR})
